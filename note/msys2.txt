pacman -S base-devel
pacman -S mingw-w64-x86_64-toolchain
pacman -S mingw64/mingw-w64-x86_64-cmake
pacman -S msys/vim

*. clean package cache
pacman -Sc
*. search package 
pacman -Ss <pacakge>
*. list package file
pacman -Ql <pacakge>

*. glfw
mkdir build
cd build
cmake -DCMAKE_INSTALL_PREFIX=<prefix> -G "MSYS Makefiles" ..
make && make install

*. glew
make install GLEW_PREFIX=<prefix> GLEW_DEST=<prefix>

*. soil
make install LOCAL=<prefix>

*. glm
mkdir build
cd build
cmake -DCMAKE_INSTALL_PREFIX=<prefix> -G "MSYS Makefiles" ..
make install

*. assimp
cmake -DBUILD_SHARED_LIBS=OFF -DASSIMP_BUILD_SAMPLES=OFF
      -DASSIMP_BUILD_TESTS=OFF
      -DCMAKE_INSTALL_PREFIX=<prefix> -G "MSYS Makefiles"
make install
