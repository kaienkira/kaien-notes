*. 刻写iso文件
    hdiutil convert -format UDRW -o 1.dmg 1.iso
    diskutil list
    diskutil unmountDisk /dev/rdiskX
    sudo dd if=1.dmg of=/dev/rdiskX bs=1m

*. 删除启动台图标
    sqlite3 $(find /private/var/folders -name com.apple.dock.launchpad 2>/dev/null)/db/db "DELETE FROM apps WHERE title='APPNAME';" && killall Dock
