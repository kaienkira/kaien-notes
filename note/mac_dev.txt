*. Homebrew
    export ALL_PROXY=socks5://127.0.0.1:9050
    /bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
    brew upgrade
*. Profile-bash
    export PS1='\u@\h:\w\$ '
*. Profile-zsh
    export LANG=zh_CN.UTF-8
    export PROMPT='%n@%m:%~$ '
*. libimobiledevice
    brew uninstall -f libimobiledevice ideviceinstaller usbmuxd
    brew install -v --HEAD --build-from-source usbmuxd libimobiledevice ideviceinstaller
    chmod 777 /var/db/lockdown
    # 查看日志
    idevicesyslog
    # 列出设备
    idevice_id -l
*. ios-deploy
    brew install npm
    npm install -g ios-deploy
    ios-deploy -b <bundle_app>
*. xcodeproj
    gem install xcodeproj
*. 切换xcode环境
    sudo xcode-select -s /Applications/Xcode.app/Contents/Developer
    sudo xcode-select -s /Library/Developer/CommandLineTools
*. 解锁keychain
    security unlock-keychain ~/Library/Keychains/login.keychain-db
    security show-keychain-info ~/Library/Keychains/login.keychain-db
*. ProvisionProfile路径
    ~/Library/MobileDevice/Provisioning Profiles
*. 解密ProvisionProfile文件
    security cms -D -i <file>
*. 系统信息
    sw_vers
    system_profiler -detailLevel basic
*. SPM(swift package manager)
    rm -rf ~/Library/org.swift.swiftpm/*
    rm -rf ~/Library/Caches/org.swift.swiftpm/*
    export all_proxy=socks5://<proxy>
    xcodebuild -resolvePackageDependencies -scmProvider system
