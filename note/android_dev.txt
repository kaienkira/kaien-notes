# 32位程序支持
    sudo apt-get install libc6:i386 libstdc++6:i386 lib32z1

# 代理设置
    vi ~/.android/androidtool.cfg
      http.proxyHost=127.0.0.1
      http.proxyPort=8118

# 更新sdk
    android list sdk
    android update sdk --no-ui --filter <package>

# 编写udev规则
    vi /etc/udev/rules.d/51-android.rules
      SUBSYSTEM=="usb", ATTR{idVendor}=="22b8", MODE="0666", GROUP="plugdev"

# 新建项目
    android list target
    android create project
        --target <target_id>
        --name <proj_name>
        --path <proj_path>
        --activity MainActivity
        --package <package_name>
# 编译运行
    ant debug
    ant installd(adb install bin/<proj_name>.apk)

# 通过网络adb
    device:
        su
        setprop service.adb.tcp.port 5555
        stop adbd
        start adbd
    pc:
        adb tcpip 5555
        adb connect <device_ip>:5555
