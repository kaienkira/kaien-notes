# v1.3 build
docker pull vyos/vyos-build:equuleus
git clone https://github.com/vyos/vyos-build
git checkout equuleus
cd vyos-build && \
    docker run --rm -it --privileged -v $(pwd):/vyos -w /vyos vyos/vyos-build:equuleus bash
./configure --architecture amd64 --build-type release
sudo make iso

# install
install image

# config
configure

set service ssh port '22'
set interfaces ethernet eth0 address dhcp
set interfaces ethernet eth0 description 'WAN'
set interfaces ethernet eth1 address '192.168.4.1/24'
set interfaces ethernet eth1 description 'LAN'

set service dhcp-server shared-network-name LAN subnet 192.168.4.0/24 default-router '192.168.4.1'
set service dhcp-server shared-network-name LAN subnet 192.168.4.0/24 name-server '192.168.4.1'
set service dhcp-server shared-network-name LAN subnet 192.168.4.0/24 lease '86400'
set service dhcp-server shared-network-name LAN subnet 192.168.4.0/24 range 0 start '192.168.4.101'
set service dhcp-server shared-network-name LAN subnet 192.168.4.0/24 range 0 stop '192.168.4.254'

set service dns forwarding cache-size '0'
set service dns forwarding listen-address '192.168.4.1'
set service dns forwarding allow-from '192.168.4.0/24'

commit
save

show interfaces ethernet eth0
