*. 修改存储目录
    export JENKINS_HOME=<jenkins_data_dir>
*. 不杀死子进程
    BUILD_ID=DONTKILLME <cmd>
*. 建立服务
    vi /opt/jenkins/bin/run_jenkins.sh
        ```
        #!/bin/bash

        export JENKINS_HOME=/opt/jenkins/data
        /usr/bin/java -jar /opt/jenkins/bin/jenkins.war --httpPort=8080
        ```
    vi /usr/lib/systemd/system/jenkins.service
        ```
        [Unit]
        Description=Jenkins

        [Service]
        ExecStart=/bin/bash /opt/jenkins/bin/run_jenkins.sh
        User=reborn
        LimitCORE=infinity

        [Install]
        WantedBy=multi-user.target
        ```
