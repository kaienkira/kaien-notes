*. network
    vi /etc/wpa_supplicant/wpa_supplicant.conf
        country=US

*. temperature monitor
    sudo vcgencmd measure_temp

*. vnc
    sudo apt-get install realvnc-server realvnc-viewer
    vncserver -Encryption PreferOff -Authentication VncAuth

*. 5inch HDMI(MPI5001)
    sudo vi /boot/config.txt
        hdmi_force_edid_audio=1
        max_usb_current=1
        hdmi_force_hotplug=1
        config_hdmi_boost=7
        hdmi_group=2
        hdmi_mode=1
        hdmi_mode=87
        hdmi_drive=1
        hdmi_cvt 800 480 60 6 0 0 0

*. bluetooth device
    sudo bluetoothctl
    scan on
    devices
    connect to <device_id>
